method public fetchAll

Returns an array containing all of the result set rows

Param

type $style , numeric=numeric keys, object=object, array=array

                                    
    public function fetchAll($as = 'array', $arguments = '')
    {

        $rows = array();
        $style = array(
            "array" => "fetchAssoc",
            "numeric" => "fetch",
            "object" => "fetchObject"
        );

        $as = !array_key_exists((string)$as, $style) ? "array" : (string)$as;

        //Getting all as an array
        while ($row = $this->$style[$as]($arguments)) {
            $rows[] = $row;
        }

        //Return the resultset;
        return $rows;
    }
