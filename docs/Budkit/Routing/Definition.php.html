                                    public function matches(Request $request, $strict = true)
    {

        $this->debug = [];
        $this->params = [];
        $this->score = 0;
        $this->failed = null;


        if ($this->isMatch($request->getPathInfo(), $request->getServer())) {
            $this->setParams();

            return true;
        }

        //If no match store reason for failure in debug var
        return false;
